attribute vec3 vertexPosition;
attribute vec2 vertexTexCoord;

uniform mat4 perspectiveMatrix;
uniform vec3 camPos;
varying vec2 texCoord;

void main(void)
{
  texCoord = vertexTexCoord;
  gl_Position = perspectiveMatrix * vec4(vertexPosition + camPos, 1.0); //* vec4(vertexPosition + vec3(0.0, 0.0, distance(vertexTexCoord, vec2(0.5, 0.5))), 1.0);
}
